library(patchwork)
library(p8105.datasets)
data("ny_noaa")
library(plotly)
ny_noaa =
  ny_noaa |>
  drop_na(tmin, tmax, prcp, snow) |>
  mutate(
    year = year(date),
    month = month(date)) |>
  filter(year == 2010)

Scatterplot

ny_noaa |>
  filter(month == 1) |>
  mutate(
    label = str_c("Prcp", prcp, "\nSnow: ", snow)
  ) |>
  plot_ly(
    x = ~tmin, y = ~tmax, color = ~snow, alpha = 0.5,
    text = ~ label,
    type = "scatter", mode = "markers")

Bar Plot

ny_noaa |>
  filter(prcp < 150) |>
  plot_ly(
    x = ~month, y = ~prcp, color = ~month,
    type = "box", colors = "viridis"
